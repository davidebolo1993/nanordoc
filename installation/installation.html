# Installation

# Dependencies

NanoR depends on the following R libraries (versions):

  • ggplot2 (>= 2.2.1)
  • reshape2 (>= 1.4.3)
  • RColorBrewer (>= 1.1.2)
  • scales (>= 0.5.0)
  • gridExtra (>= 2.3)
  • rhdf5 (>= 2.14)
  • ShortRead (>= 1.24.0)
  • data.table (>= 1.10.4.3)

Most of the dependencies above are available through the Comprehensive R Archive Network.

Cpak <- function(pkg){
    new.pkg <- pkg[!(pkg %in% installed.packages()[, "Package"])]
    if (length(new.pkg)) 
        install.packages(new.pkg, dependencies = TRUE, repos="http://cran.cnr.berkeley.edu/")
    sapply(pkg, require, character.only = TRUE)
}

packages <- c("ggplot2","reshape2","RColorBrewer","scales","gridExtra","data.table")
Cpak(packages)

Dependencies 'rhdf5' and 'ShortRead' must be installed through Bioconductor.

Bpak<- function(pkg){
    new.pkg <- pkg[!(pkg %in% installed.packages()[, "Package"])]
    if (length(new.pkg))
        if (!requireNamespace("BiocManager", quietly = TRUE))
            install.packages("BiocManager")
        BiocManager::install(new.pkg)
    sapply(pkg, require, character.only = TRUE)
}

packages<-c("rhdf5", "ShortRead")
Bpak(packages)

# Installation from source

Once the prerequisites above are met, NanoR can be cloned.

git clone https://github.com/davidebolo1993/NanoR.git

Eventually, the cloned package can be installed.

install.packages("/path/to/NanoR/NanoR.tar.gz", repos=NULL)
